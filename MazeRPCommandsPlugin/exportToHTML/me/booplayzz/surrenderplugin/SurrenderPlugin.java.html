package me.booplayzz.surrenderplugin;

import org.bukkit.ChatColor;
import org.bukkit.command.Command;
import org.bukkit.command.CommandSender;
import org.bukkit.entity.Player;
import org.bukkit.plugin.java.JavaPlugin;

import java.io.Console;

public final class SurrenderPlugin extends JavaPlugin {

    @Override
    public void onEnable() {
        // Plugin startup logic

    }

    @Override
    public boolean onCommand(CommandSender sender, Command command, String label, String[] args) {

        if (command.getName().equalsIgnoreCase("surrender")){

            Player p = (Player) sender;

            p.setHealth(0.0);
            p.sendMessage(ChatColor.GOLD + "[" + ChatColor.GREEN + "MazeRP" + ChatColor.GOLD + "] " + ChatColor.GOLD + "You have surrendered to the maze.");
        }


        return true;
    }
}
